def print(msg)
  under_shell = ::Warp::Dir.eval_context?
  if msg == ' '
    under_shell ? stream.printf(%Q{printf '\\n'; }) : stream.printf("\n")
  else
    msg.split("\n").each do |line|
      under_shell ? stream.printf(%Q{printf -- '#{line.gsub(/['`]/, '\"')}\\n';}) : stream.printf("#{line}\n")
    end
  end
end
